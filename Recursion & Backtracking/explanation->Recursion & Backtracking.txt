    int Pascal (ll x ,ll y) {
        if (x==y||y==0) return 1;

        return Pascal(x-1,y)+Pascal(x-1,y-1);
    }

    int GCD(int x,int y) {
        if (y>x)swap(x,y);
        if (y==0)return x;
        return GCD(x-y,y);
    }
    int FAST_GCD(int x,int y) {
        if (y==0)return x;
        return FAST_GCD(y,x%y);
    }
// fast power iterative ------------------------------------------------------------------------------
    ll fast_power_mod(ll x,ll y,ll mod) { // without rec   and to as rec     /// time log (y)
        ll res=1;
        while (y) {
            if (y&1) {
                    res=(res*x)%mod;
            }
            x=(x*x)%mod;
            y/=2;
        }
        return  res;
    }
// fast power Recurcion-----------------------------------------------------------------------------------
ll fast_power_mod_Recurcion(ll base, ll power,ll mod) {
        if ( power<=0)return 1LL;
        ll res=fast_power_mod((base*base)%mod,power/2,mod)%mod;
        if (power&1) res=(res*base)%mod;
        return  res;
    }

int n;
void rec(string s,int n1,int n2) {
    if (s.size()==n*2) {
        if (n1==n2) {
            cout<<s<<endl;
        }
        return;
    }
    rec(s+'0',n1,n2);
    if (s.size()<n)rec(s+'1',n1+1,n2);
    else rec(s+'1',n1,n2+1);
}
