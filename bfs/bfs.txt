int N,M;
void bfs(int s) {
    queue<int>q;
    q.push(s);
    vis[s]=1;
    level[s]=0;
    while (!q.empty()) {
        int cur=q.front();
        q.pop();
        element.push_back(cur);
        for (auto it:adj[cur]) {
            if (!vis[it]) {
                q.push(it);
                vis[it]=1;
                level[it]=level[cur]+1;
            }
        }
    }

}
